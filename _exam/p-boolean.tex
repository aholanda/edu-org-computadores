
\def\MIN{
  \question[1]
  Simplificar as expressões Booleanas:
  
  \begin{parts}
    \part
    $\!{ \!{{(s\+\!t)}} \+ \!{(s\+\!u)} \+ v}$

    \begin{solution}
      $(s + \!t)(s + \!u) \!v$
    \end{solution}

    \part
    $\!xyz \+ x\!yz \+ xy\!z \+ xyz$
    \begin{solution}
      $yz+xz+xy$
    \end{solution}

    \part
    $\!x(x\+y) \+ (y\+x)(x\+\!y)$
    \begin{solution}
      $x+y$
    \end{solution}


    \part
    $\!{ \!{x\+yz}\ {\!{x\!y}} }$
    \begin{solution}
      $x+yz$
    \end{solution}

  \end{parts}
}
